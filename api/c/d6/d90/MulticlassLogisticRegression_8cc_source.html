94  {

99  REQUIRE(m_labels, "%s::train_machine(): No labels attached!\n");
100  REQUIRE(m_labels->get_label_type()==LT_MULTICLASS, "%s::train_machine(): "
101  "Attached labels are no multiclass labels\n");
102  REQUIRE(m_multiclass_strategy, "%s::train_machine(): No multiclass strategy"
103  " attached!\n");
104 
105  int32_t n_classes = ((CMulticlassLabels*)m_labels)->get_num_classes();
106  int32_t n_feats = m_features->get_dim_feature_space();
107 
108  slep_options options = slep_options::default_options();
109  if (m_machines->get_num_elements()!=0)
110  {
111  SGMatrix<float64_t> all_w_old(n_feats, n_classes);
112  SGVector<float64_t> all_c_old(n_classes);
113  for (int32_t i=0; i<n_classes; i++)
114  {
115  CLinearMachine* machine = (CLinearMachine*)m_machines->get_element(i);
116  SGVector<float64_t> w = machine->get_w();
117  for (int32_t j=0; j<n_feats; j++)
118  all_w_old(j,i) = w[j];
119  all_c_old[i] = machine->get_bias();
120  SG_UNREF(machine);
121  }
122  options.last_result = new slep_result_t(all_w_old,all_c_old);
123  m_machines->reset_array();
124  }
125  options.tolerance = m_epsilon;
126  options.max_iter = m_max_iter;
127  slep_result_t result = slep_mc_plain_lr(m_features,(CMulticlassLabels*)m_labels,m_z,options);
128 
129  SGMatrix<float64_t> all_w = result.w;
130  SGVector<float64_t> all_c = result.c;
131 
132  for (int32_t i=0; i<n_classes; i++)
133  {
134  SGVector<float64_t> w(n_feats);
135  for (int32_t j=0; j<n_feats; j++)
136  w[j] = all_w(j,i);
137 
138  float64_t c = all_c[i];
139  CLinearMachine* machine = new CLinearMachine();
140  machine->set_w(w);
141  machine->set_bias(c);
142  m_machines->push_back(machine);
143  }
144  return true;
145  }
