164  {
165  Eigen::PermutationMatrix<Dynamic,Dynamic> perm(o->X.cols());
166  perm.setIdentity();
167  r.shuffle(perm.indices().data(), perm.indices().data()+perm.indices().size());
168  o->X = o->X * perm; // shuffles columns of X
169  o->y = (o->y.transpose() * perm).transpose() ; // shuffle y too
170 
171  if(o->Z.size() > 0)
172  {
173  std::vector<int> zidx(o->y.size());
174  std::iota(zidx.begin(), zidx.end(), 0);
175  VectorXi zw = Map<VectorXi>(zidx.data(), zidx.size());
176  // shuffle z indices

178  // assign shuffled zw to zidx
179  zidx.assign(zw.data(), zw.data() + zw.size());
180  for(auto &val : o->Z)

357  void DataRef::reorder_longitudinal(vector<ArrayXf> &vec1, const vector<int>& order)
358  {
359  vector<int> index = order;
360  // Fix all elements one by one
361  for (int i=0; i<index.size(); i++)
362  {
363  // While index[i] and vec1[i] are not fixed
364  while (index.at(i) != i)
365  {
366  // Store values of the target (or correct)
367  // position before placing vec1[i] there
368  int oldTargetI = index.at(index.at(i));
369  auto oldTargetE = vec1.at(index.at(i));
370 
371  // Place vec1[i] at its target (or correct)
372  // position. Also copy corrected index for
373  // new position
374  vec1.at(index.at(i)) = vec1.at(i);
375  index.at(index.at(i)) = index.at(i);
376 
377  // Copy old target values to vec1[i] and
378  // index[i]
379  index.at(i) = oldTargetI;
380  vec1.at(i) = oldTargetE;
381  }
382  }
