296  MatrixXf Individual::state_to_phi(State& state)
297  {
298  // we want to preserve the order of the outputs in the program
299  // in the order of the outputs in Phi.
300  // get root output types
301  this->dtypes.clear();
302  for (auto r : program.roots())
303  {
304  this->dtypes.push_back(program.at(r)->otype);
305  }
306  // convert state_f to Phi
307  logger.log("converting State to Phi",3);
308  int cols;
309 
310  if (state.f.size()==0)
311  {
312  if (state.c.size() == 0)
313  {
314  if (state.b.size() == 0)
315  HANDLE_ERROR_THROW("Error: no outputs in State");
316 
317  cols = state.b.top().size();
318  }
319  else{
320  cols = state.c.top().size();
321  }
322  }
323  else{
324  cols = state.f.top().size();
325  }
326 
327  // define Phi matrix
328  Matrix<float,Dynamic,Dynamic,RowMajor> Phi (
329  state.f.size() + state.c.size() + state.b.size(),
330  cols);
331  ArrayXf Row;
332  std::map<char,int> rows;
333  rows['f']=0;
334  rows['c']=0;
335  rows['b']=0;
336 
337  // add rows (features) to Phi with appropriate type casting
338  for (int i = 0; i < this->dtypes.size(); ++i)
339  {
340  char rt = this->dtypes.at(i);
341 
342  switch (rt)
343  {
344  case 'f':
345  // add state_f to Phi
346  Row = ArrayXf::Map(state.f.at(rows.at(rt)).data(),cols);
347  break;
348  case 'c':
349  // convert state_c to Phi
350  Row = ArrayXi::Map(
351  state.c.at(rows.at(rt)).data(),cols).cast<float>();
352  break;
353  case 'b':
354  // add state_b to Phi
355  Row = ArrayXb::Map(
356  state.b.at(rows.at(rt)).data(),cols).cast<float>();
357  break;
358  default:
359  HANDLE_ERROR_THROW("Unknown root type");
360  }
361  // remove nans, set infs to max and min
362  clean(Row);
363  Phi.row(i) = Row;
364  ++rows.at(rt);
365  }
366  return Phi;
367  }

867  int Individual::check_dominance(const Individual& b) const
868  {
869  /* Check whether this individual dominates b.
870  *
871  * Input:
872  *
873  * b: another individual
874  *
875  * Output:
876  *
877  * 1: this individual dominates b; -1: b dominates this;
878  * 0: neither dominates
879  */
880 
881  int flag1 = 0, // to check if this has a smaller objective
882  flag2 = 0; // to check if b has a smaller objective
883 
884  for (int i=0; i<obj.size(); ++i) {
885  if (obj.at(i) < b.obj.at(i))
886  flag1 = 1;
887  else if (obj.at(i) > b.obj.at(i))
888  flag2 = 1;
889  }
890 
891  if (flag1==1 && flag2==0)
892  // there is at least one smaller objective for this and none
893  // for b
894  return 1;
895  else if (flag1==0 && flag2==1)
896  // there is at least one smaller objective for b and none
897  // for this
898  return -1;
899  else
900  // no smaller objective or both have one smaller
901  return 0;
902
